  \end{itemize}

\begin{frame}{Überschriften müssen informativ sein.}

  Man kann Overlays erzeugen\dots
  \begin{itemize}
  \item mit dem \texttt{pause}-Befehl:
    \begin{itemize}
    \item
      Erster Punkt.
      \pause
    \item    
      Zweiter Punkt.
    \end{itemize}
  \item
    mittels Overlay-Spezifikationen:
    \begin{itemize}
    \item<3->
      Erster Punkt.
    \item<4->
      Zweiter Punkt.
    \end{itemize}
  \item
    mit dem allgemeinen \texttt{uncover}-Befehl:
    \begin{itemize}
      \uncover<5->{\item
        Erster Punkt.}
      \uncover<6->{\item
        Zweiter Punkt.}
    \end{itemize}
  \end{itemize}
\end{frame}
