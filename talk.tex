\begin{frame}[fragile]{Beispiel unoptimiert}
\begin{verbatim}
for (uint8_t t = 0; t < 100; t++) {
  for (uint8_t x = 0; x < 16; x++) {
    for (uint8_t y = 0; y < 16; y++) {
      buffer[x][y] = 566*t*sin(y*y) + t*t*x*x*y/300 +
                     3*t*t;
    }
  }
  ausgabe(buffer);
  delay(10);	
}
\end{verbatim}
\end{frame}

\begin{frame}[fragile]{Optimierung Schritt 1}
\begin{verbatim}
for (uint8_t t = 0; t < 100; t++) {
  uint16_t h1 = 566*t, h2 = 3*t*t;
  for (uint8_t x = 0; x < 16; x++) {
    uint16_t h3 = t*t*x*x;
    for (uint8_t y = 0; y < 16; y++) {
      buffer[x][y] = h1*sin(y*y) + h3*y/300 + h2;
    }
  }
  ausgabe(buffer);
}
\end{verbatim}
\end{frame}
